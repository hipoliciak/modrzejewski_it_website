## Why Bulk-Safe Triggers Matter

Salesforce governor limits exist to ensure fair resource usage on the multi-tenant platform. A trigger that works perfectly for a single record can catastrophically fail during a data load of 10,000 records. Every Apex trigger **must** be designed to handle 200 records in a single execution context.

## The Trigger Handler Framework

Avoid business logic directly in trigger files. Instead, use the handler pattern:

```apex
// Trigger file: OpportunityTrigger.trigger
trigger OpportunityTrigger on Opportunity (
    before insert, before update, before delete,
    after insert, after update, after delete, after undelete
) {
    OpportunityTriggerHandler handler = new OpportunityTriggerHandler();
    
    if (Trigger.isBefore) {
        if (Trigger.isInsert) handler.onBeforeInsert(Trigger.new);
        if (Trigger.isUpdate) handler.onBeforeUpdate(Trigger.new, Trigger.oldMap);
        if (Trigger.isDelete) handler.onBeforeDelete(Trigger.old);
    }
    
    if (Trigger.isAfter) {
        if (Trigger.isInsert) handler.onAfterInsert(Trigger.new);
        if (Trigger.isUpdate) handler.onAfterUpdate(Trigger.new, Trigger.oldMap);
        if (Trigger.isDelete) handler.onAfterDelete(Trigger.oldMap);
        if (Trigger.isUndelete) handler.onAfterUndelete(Trigger.new);
    }
}
```

## The Handler Class

```apex
// OpportunityTriggerHandler.cls
public with sharing class OpportunityTriggerHandler {
    
    public void onBeforeInsert(List<Opportunity> newOpps) {
        setDefaultValues(newOpps);
        validateRequiredFields(newOpps);
    }
    
    public void onAfterInsert(List<Opportunity> newOpps) {
        createRelatedRecords(newOpps);
        sendNotifications(newOpps);
    }
    
    public void onBeforeUpdate(
        List<Opportunity> newOpps, 
        Map<Id, Opportunity> oldOppMap
    ) {
        validateStageTransitions(newOpps, oldOppMap);
    }
    
    public void onAfterUpdate(
        List<Opportunity> newOpps, 
        Map<Id, Opportunity> oldOppMap
    ) {
        List<Opportunity> closedWonOpps = new List<Opportunity>();
        
        for (Opportunity opp : newOpps) {
            Opportunity oldOpp = oldOppMap.get(opp.Id);
            // Only process records that actually changed stage
            if (opp.StageName == 'Closed Won' && oldOpp.StageName != 'Closed Won') {
                closedWonOpps.add(opp);
            }
        }
        
        if (!closedWonOpps.isEmpty()) {
            processClosedWonOpportunities(closedWonOpps);
        }
    }
    
    // Private methods - business logic lives here
    private void setDefaultValues(List<Opportunity> opps) {
        for (Opportunity opp : opps) {
            if (opp.LeadSource == null) {
                opp.LeadSource = 'Web';
            }
            if (opp.ForecastCategoryName == null) {
                opp.ForecastCategoryName = 'Pipeline';
            }
        }
    }
    
    private void createRelatedRecords(List<Opportunity> opps) {
        // Collect all Account IDs for a single query
        Set<Id> accountIds = new Set<Id>();
        for (Opportunity opp : opps) {
            if (opp.AccountId != null) {
                accountIds.add(opp.AccountId);
            }
        }
        
        // One SOQL query for all accounts
        Map<Id, Account> accountMap = new Map<Id, Account>(
            [SELECT Id, OwnerId, Type FROM Account WHERE Id IN :accountIds]
        );
        
        // Build records in memory first, then DML once
        List<Task> tasksToInsert = new List<Task>();
        for (Opportunity opp : opps) {
            if (opp.AccountId != null && accountMap.containsKey(opp.AccountId)) {
                Account acc = accountMap.get(opp.AccountId);
                tasksToInsert.add(new Task(
                    Subject = 'Follow up on new opportunity',
                    WhatId = opp.Id,
                    OwnerId = opp.OwnerId,
                    ActivityDate = Date.today().addDays(3),
                    Status = 'Not Started',
                    Priority = 'Normal'
                ));
            }
        }
        
        if (!tasksToInsert.isEmpty()) {
            insert tasksToInsert; // Single DML statement for all records
        }
    }
    
    private void validateRequiredFields(List<Opportunity> opps) {
        for (Opportunity opp : opps) {
            if (opp.CloseDate < Date.today()) {
                opp.CloseDate.addError('Close Date cannot be in the past.');
            }
            if (opp.Amount != null && opp.Amount < 0) {
                opp.Amount.addError('Amount must be a positive value.');
            }
        }
    }
    
    private void validateStageTransitions(
        List<Opportunity> newOpps, 
        Map<Id, Opportunity> oldOppMap
    ) {
        Set<String> closedStages = new Set<String>{'Closed Won', 'Closed Lost'};
        
        for (Opportunity opp : newOpps) {
            Opportunity oldOpp = oldOppMap.get(opp.Id);
            // Prevent reopening closed opportunities without a field set
            if (closedStages.contains(oldOpp.StageName) 
                && !closedStages.contains(opp.StageName)
                && opp.Description == null) {
                opp.StageName.addError(
                    'A reason is required in the Description field to reopen a closed opportunity.'
                );
            }
        }
    }
    
    private void processClosedWonOpportunities(List<Opportunity> closedWonOpps) {
        // Process closed won logic - create orders, notify teams, etc.
        List<Id> oppIds = new List<Id>();
        for (Opportunity opp : closedWonOpps) {
            oppIds.add(opp.Id);
        }
        // Enqueue async job to avoid hitting limits
        System.enqueueJob(new ClosedWonOpportunityQueueable(oppIds));
    }
    
    private void sendNotifications(List<Opportunity> opps) {
        // Notification logic
    }
}
```

    enabled: true
    bypass_permission: "Bypass_OpportunityTrigger"

  batch_size: 200

## Key Rules to Remember

1. **Never query or DML inside a loop** — collect IDs, query once outside the loop
2. **Use Maps for record lookup** — `Map<Id, SObject>` pattern for O(1) access
3. **Bulkify everything** — design for 200 records minimum
4. **One trigger per object** — use a handler/dispatcher pattern
5. **Use Sets for deduplication** — avoid processing the same record twice

## Testing for Bulk Safety

```apex
@isTest
private class OpportunityTriggerHandlerTest {
    
    @TestSetup
    static void makeData() {
        Account testAccount = new Account(Name = 'Test Corp', Type = 'Customer');
        insert testAccount;
    }
    
    @isTest
    static void testBulkInsert_200Records() {
        Account acc = [SELECT Id FROM Account LIMIT 1];
        
        List<Opportunity> opps = new List<Opportunity>();
        for (Integer i = 0; i < 200; i++) {
            opps.add(new Opportunity(
                Name = 'Test Opp ' + i,
                AccountId = acc.Id,
                StageName = 'Prospecting',
                CloseDate = Date.today().addDays(30),
                Amount = 10000 * (i + 1)
            ));
        }
        
        Test.startTest();
        insert opps; // This will trigger bulk processing
        Test.stopTest();
        
        List<Task> createdTasks = [SELECT Id FROM Task WHERE WhatId IN :opps];
        System.assertEquals(200, createdTasks.size(), 'Should create one task per opportunity');
    }
}
```
